#### 1 Write a Shell script to accept a string as command line argument and reverse the same.

```bash copy
#!/bin/bash

# Get the input string
string=$1

# Reverse the string
reversed_string=$(echo $string | rev)

# Print the result
echo "The reversed string of '$string' is '$reversed_string'"

```

#### 3 Write a shell script to calculate the loss percentage of an article, given the cost price and the selling price as command line arguments.

```bash copy
#!/bin/bash

# Get the cost price
cost_price=$1

# Get the selling price
selling_price=$2

# Calculate the loss
loss=$(echo "($cost_price - $selling_price) / $cost_price * 100" | bc -l)

# Print the result
echo "The loss percentage is $loss%"

```

#### 3 Write a shell script to accept the name of the user and check out if the same has logged in or not

```bash copy
#!/bin/bash

# Get the input user name
username=$1

# Check if the user is logged in
logged_in=$(who | grep $username | wc -l)

# Print the result
if [ $logged_in -eq 0 ]; then
  echo "User '$username' is not logged in."
else
  echo "User '$username' is logged in."
fi

```

#### 4 Write a shell script to check whether the file whose name is scanned exists and readable.

```bash copy
#!/bin/bash

# Get the input file name
file=$1

# Check if the file exists
if [ ! -f $file ]; then
    echo "File not found."
    exit 1
fi

# Check if the file is readable

    echo "File is not readable."
    exit 1
fi

echo "File exists and is readable."
```

#### 6 . Write a shell script to accept a number and a word as command line arguments and print the word the given number of times on each line.

```bash copy
#!/bin/bash

# Get the input number
num=$1

# Get the input word
word=$2

  echo $word
done

```

#### 7 Write a shell script to find the file or directory with the maximum size in the current directory

```bash copy
#!/bin/bash

# Initialize the max_size variable
max_size=0

# Initialize the max_file variable
max_file=""

# Iterate through all files and directories in the current directory
for file in *
do
    # Get the size of the current file/directory
    size=$(du -s $file | awk '{print $1}')

    # Check if the size is greater than the current max size
    if [ $size -gt $max_size ]; then
        max_size=$size
        max_file=$file
    fi
done

# Print the result
echo "The largest file/directory in the current directory is $max_file with a size of $max_size KB"

```

#### 8 Write a shell script to accept two filenames and check if both exist. If the second filename exists, then the contents of the first filename should be appended to it. If the second filename does not exist then create a newfile with the contents of the first file

```bash copy
#!/bin/bash

first_file=$1
second_file=$2

if [ -f "$first_file" ]; then
  if [ -f "$second_file" ]; then
    cat "$first_file" >> "$second_file"
    echo "Appended contents of $first_file to $second_file"
  else
    cp "$first_file" "$second_file"
    echo "Created new file $second_file with contents of $first_file"
  fi
else
  echo "Error: $first_file does not exist"
fi

```

#### 9 Write a shell script to accept a number in the command line and displays the sum up to that number. By default, the sum up to 50 should be displayed.

```bash copy
#!/bin/bash

if [ $# -eq 0 ]; then
    number=50
else
    number=$1
fi

sum=0
for ((i=1;i<=number;i++))
do
    sum=$((sum + i))
done

echo "The sum up to $number is $sum"

```

#### 10 Write a shell script to find the number of ordinary files and directory files in the current directory

```bash copy
#!/bin/bash

num_files=0
num_dirs=0

for file in *; do
  if [ -f "$file" ]; then
    num_files=$((num_files + 1))
  elif [ -d "$file" ]; then
    num_dirs=$((num_dirs + 1))
  fi
done

echo "Number of files: $num_files"
echo "Number of directories: $num_dirs"
```

#### 11 Write a shell script to accept an alphabet from the user and list all the files/directory starting with that alphabet in the current directory

```bash copy
#!/bin/bash

echo "Enter an alphabet: "
read alphabet

echo "Files/directories starting with $alphabet:"

for file in *; do
  if [[ "$file" == "$alphabet"* ]]; then
    echo "$file"
    if [ -x "$file" ]; then
      ./"$file"
    fi
  fi
done

```
