l1= list(input("Enter variables: ").split())
n=len(l1)-1
print(n)
print("No. of list is:-",l1)

def sorting(l1,n):
    temp=[]
    for i in range(0,n):
        for j in range(0,n):
            if l1[j]>l1[j+1]:
                temp.append(l1[j])
                l1[j]=l1[j+1]
                l1[j+1]=temp[0]
                temp.pop(0)
    print("After sorting list:-",l1)
sorting(l1,n)

### 5 ‚¶Å	Write a function to add two lists of the same length term-by-term & return new list 
```py filename="P6_5.py"   copy
def listadd(a, b):
    c = []
    for i in range(0, len(a)):
        c.append(a[i]+b[i])
    return c
ans = listadd([11,12,13], [21,22,23])
print("Addition of two lists:-",ans)
```
